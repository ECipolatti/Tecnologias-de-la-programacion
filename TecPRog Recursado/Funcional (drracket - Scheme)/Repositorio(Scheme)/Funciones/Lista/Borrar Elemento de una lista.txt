(define borrar-list
  (lambda (elemento lista)
    (if (null? lista)
        '()
        (if (eqv? elemento (car lista))
            (cons (car(cdr lista))(borrar-list elemento (cdr(cdr lista))))
            (cons (car lista)(borrar-list elemento (cdr lista)))
            )
        )
    )
  )
(borrar-list 9 '(1 2 9 5 6 4 9 7) )
