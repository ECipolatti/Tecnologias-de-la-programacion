(define eliminaELEM
  (lambda (elemento lista)
    (if (null? lista)
        '()
        (if (eqv? elemento (car lista))
            (eliminaELEM elemento (cdr lista))
            (if (list? (car lista))
                (cons (eliminaELEM elemento (car lista))(eliminaELEM elemento(cdr lista)))
                 (cons (car lista) (eliminaELEM elemento(cdr lista)))
             )
        )
    )
  )
  )
(eliminaELEM 2 '((1 2 3) 2 7 (1 3 (6 2 8) 0 2)))
